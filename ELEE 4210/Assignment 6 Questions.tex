\\
\\

\\

\begin{flalign}
    y(t) &= x(t)*h(t) = \int_{-\infty}^{\infty} x(t-\tau)h(\tau)\,d\tau.&& \nonumber
\end{flalign}
\noindent Given that the system is causal,
\begin{flalign}
    y(t) &= u(t)\int_{0}^{t} u(t-\tau)e^{-t+\tau}u(\tau)e^{-\tau}\,d\tau&& \nonumber \\

\end{flalign}
